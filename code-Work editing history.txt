// เชื่อมต่อฐานข้อมูล

    die("Connection failed: " . $conn->connect_error);
}

$sql = "SELECT edited_at, action, edited_by FROM edit_history ORDER BY edited_at DESC";
$result = $conn->query($sql);
?>

<!DOCTYPE html>
